## 算法

二叉树、二分查找

```js
// 当时用的 js，用 ts 只是为了表达的更清晰一点
interface Barrage {
	time: Number;
	barrage: String;
}
const binarySearch = (arr: Barrage[], time: number) => {
	let l = 0
	let r = arr.length - 1
	while (l <= r) {
		const mid = (l + r) / 2
		if (arr[mid].time === time) {
			return arr[mid].barrage
		} else if (arr[mid].time < time) {
			l = mid + 1
		} else {
			r = mid - 1
		}
	}
	return null
}
```
